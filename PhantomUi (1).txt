local function ExecuteScript(script)
	local f, err = loadstring(script)
	if f then
		f()
	else
		print("Script execution error:", err)
	end
end

	local script = Phantom.ScriptInput.Text
	ExecuteScript(script)
	game.StarterGui:SetCore("SendNotification", {
		Title = "Script";
		Text = "Script Executed";
		Duration = 3;
	})
end)

local function VWTEP_fake_script() -- Phantom.ScrollingFrame.LocalScript 
	local script = Instance.new('LocalScript', Phantom.ScrollingFrame)

	-- Reference to the parent and child ScrollingFrames
	local parentScrollingFrame = script.Parent
	local childScrollingFrame = parentScrollingFrame:WaitForChild("ChildScrollingFrame")
	
	-- Disable automatic CanvasSize adjusting for the child ScrollingFrame
	childScrollingFrame.ScrollingEnabled = false
	
	-- Function to update the size of the child ScrollingFrame when the parent size changes
	local function updateChildSize()
		local parentSize = parentScrollingFrame.AbsoluteSize
		childScrollingFrame.Size = UDim2.new(1, 0, 0, parentSize.Y)
	end
	
	updateChildSize()
	
	parentScrollingFrame:GetPropertyChangedSignal("AbsoluteSize"):Connect(updateChildSize)
end
