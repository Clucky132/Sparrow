local function ExecuteScript(script)
	local f, err = loadstring(script)
	if f then
		f()
	else
		print("Script execution error:", err)
	end
end

local function AddTextLine(text, textColor)
	local TextLabel = Instance.new("TextLabel")
	TextLabel.Size = UDim2.new(1, 0, 0, 0) -- Set initial height to 0
	TextLabel.BackgroundTransparency = 1
	TextLabel.Text = text
	TextLabel.TextWrapped = true
	TextLabel.TextXAlignment = Enum.TextXAlignment.Left
	TextLabel.TextYAlignment = Enum.TextYAlignment.Top
	TextLabel.TextSize = 14
	TextLabel.TextColor3 = textColor or Color3.new(0, 0, 0) -- Set text color to black by default
	TextLabel.Parent = Frame

	-- Function to update TextLabel size based on text bounds
	local function UpdateTextLabelSize()
		TextLabel.Size = UDim2.new(1, 0, 0, TextLabel.TextBounds.Y)
		ScrollingFrame.CanvasSize = UDim2.new(0, UIListLayout.AbsoluteContentSize.X, 0, UIListLayout.AbsoluteContentSize.Y)
	end

	-- Update size initially
	UpdateTextLabelSize()

	-- Update size whenever text content changes
	TextLabel:GetPropertyChangedSignal("Text"):Connect(UpdateTextLabelSize)

	-- Update size whenever the UIListLayout resizes
	UIListLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(UpdateTextLabelSize)
end

-- Redirect console output to the ScrollingFrame
local function ConsoleOutput(message, messageType)
	local time = os.date("%H:%M:%S") -- Get the current time
	local formattedMessage = "[" .. time .. "] " .. message

	if messageType == Enum.MessageType.MessageOutput then
		AddTextLine(formattedMessage) -- Add time prefix to the message
	elseif messageType == Enum.MessageType.MessageInfo then
		AddTextLine("[INFO] " .. formattedMessage, Color3.fromRGB(0, 255, 0)) -- Green text for INFO
	elseif messageType == Enum.MessageType.MessageWarning then
		AddTextLine("[WARNING] " .. formattedMessage, Color3.fromRGB(255, 255, 0)) -- Yellow text for WARNING
	elseif messageType == Enum.MessageType.MessageError then
		AddTextLine("[ERROR] " .. formattedMessage, Color3.fromRGB(255, 0, 0)) -- Red text for ERROR
	end
end

game:GetService("LogService").MessageOut:Connect(ConsoleOutput)
